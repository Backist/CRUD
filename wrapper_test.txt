import functools

def mydecorator(params):
    """Ponemos nombre al la funcion que se va a wrappear 
    (Con este nombre se llamara a la funcion que se va a wrappear)"""
    @functools.wraps(params)
    def wrapper_function(*args, **kwargs):
        """Ponemos nombre a la funcion que se va a wrappear"""
        return params(*args, **kwargs)+1
    return wrapper_function

@mydecorator
def suma(a, b):
    return a+b
